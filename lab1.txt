import java.math.BigDecimal;
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

//интерфейсы и сущности
interface PurchasableItem {
    void accept(PurchasableItemVisitor itemVisitor);
}

final class ProductItem implements PurchasableItem {
    private final String productId;
    private final String productName;
    private final BigDecimal baseCostValue;
    private final double weightInKilograms;

    public ProductItem(String productId, String productName, BigDecimal baseCostValue, double weightInKilograms) {
        this.productId = productId;
        this.productName = productName;
        this.baseCostValue = baseCostValue;
        this.weightInKilograms = weightInKilograms;
    }

    public BigDecimal getBaseCostValue() { return baseCostValue; }
    public double getWeightInKilograms() { return weightInKilograms; }

    @Override
    public void accept(PurchasableItemVisitor itemVisitor) {
        itemVisitor.visitProduct(this);
    }
}
//узел компоновщика
final class BundleItem implements PurchasableItem {
    private final String bundleId;
    private final List<PurchasableItem> purchasableItemList;

    public BundleItem(String bundleId, List<PurchasableItem> inputItemList) {
        this.bundleId = bundleId;
        this.purchasableItemList = List.copyOf(inputItemList);
    }

    public List<PurchasableItem> getPurchasableItemList() { return purchasableItemList; }

    @Override
    public void accept(PurchasableItemVisitor itemVisitor) {
        itemVisitor.visitBundle(this);
        for (PurchasableItem item : purchasableItemList) {
            item.accept(itemVisitor);
        }
    }
}

// стратегии
interface PricingStrategy {
    BigDecimal calculateFinalCost(BigDecimal baseCostValue);
}

final class RegularPricingStrategy implements PricingStrategy {
    @Override
    public BigDecimal calculateFinalCost(BigDecimal baseCostValue) { return baseCostValue; }
}

final class HolidayPricingStrategy implements PricingStrategy {
    private static final BigDecimal HOLIDAY_DISCOUNT_MULTIPLIER = new BigDecimal("0.85");

    @Override
    public BigDecimal calculateFinalCost(BigDecimal baseCostValue) {
        return baseCostValue.multiply(HOLIDAY_DISCOUNT_MULTIPLIER);
    }
}

// посетитли
interface PurchasableItemVisitor {
    void visitProduct(ProductItem productItem);
    void visitBundle(BundleItem bundleItem);
}

final class CostItemVisitor implements PurchasableItemVisitor {
    private final PricingStrategy pricingStrategy;
    private BigDecimal totalCostValue = BigDecimal.ZERO;

    public CostItemVisitor(PricingStrategy pricingStrategy) { this.pricingStrategy = pricingStrategy; }

    @Override
    public void visitProduct(ProductItem productItem) {
        totalCostValue = totalCostValue.add(pricingStrategy.calculateFinalCost(productItem.getBaseCostValue()));
    }

    @Override
    public void visitBundle(BundleItem bundleItem) {}

    public BigDecimal getTotalCostValue() { return totalCostValue; }
}

final class WeightItemVisitor implements PurchasableItemVisitor {
    private double totalWeightValue = 0.0;

    @Override
    public void visitProduct(ProductItem productItem) { totalWeightValue += productItem.getWeightInKilograms(); }

    @Override
    public void visitBundle(BundleItem bundleItem) { totalWeightValue += 0.5; }

    public double getTotalWeightValue() { return totalWeightValue; }
}

// строитель
final class ProductItemBuilder {
    private String productId;
    private String productName;
    private BigDecimal baseCostValue;
    private double weightInKilograms;

    public ProductItemBuilder withProductId(String productId) { this.productId = productId; return this; }
    public ProductItemBuilder withProductName(String productName) { this.productName = productName; return this; }
    public ProductItemBuilder withBaseCostValue(BigDecimal baseCostValue) { this.baseCostValue = baseCostValue; return this; }
    public ProductItemBuilder withWeightInKilograms(double weightInKilograms) { this.weightInKilograms = weightInKilograms; return this; }

    public ProductItem build() { return new ProductItem(productId, productName, baseCostValue, weightInKilograms); }
}

// фабрика
abstract class AbstractStoreFactory {
    public abstract PurchasableItem createStandardGroceryBundle();
    public abstract PurchasableItem createElectronicsBundle();
}

final class WeekendStoreFactory extends AbstractStoreFactory {
    @Override
    public PurchasableItem createStandardGroceryBundle() {
        List<PurchasableItem> list = new ArrayList<>();
        list.add(new ProductItemBuilder().withProductId("FOOD_1").withProductName("Apple").withBaseCostValue(new BigDecimal("120.50")).withWeightInKilograms(1.2).build());
        return new BundleItem("BUNDLE_100", list);
    }

    @Override
    public PurchasableItem createElectronicsBundle() {
        List<PurchasableItem> list = new ArrayList<>();
        list.add(new ProductItemBuilder().withProductId("ELEC_1").withProductName("Wireless Mouse").withBaseCostValue(new BigDecimal("1500.00")).withWeightInKilograms(0.3).build());
        return new BundleItem("BUNDLE_200", list);
    }
}

class PurchaseProcessingException extends RuntimeException {
    public PurchaseProcessingException(String message) { super(message); }
}

// главный класс
public class Main {
    public static void main(String[] args) {
        try {
            processStoreSimulation();
        } catch (PurchaseProcessingException e) {
            System.err.println("Ошибка: " + e.getMessage());
        }
    }

    private static void processStoreSimulation() {
        AbstractStoreFactory storeFactory = new WeekendStoreFactory();
        PurchasableItem groceryBundle = storeFactory.createStandardGroceryBundle();
        PurchasableItem electronicsBundle = storeFactory.createElectronicsBundle();

        BundleItem shoppingCartBundle = new BundleItem("CART_MAIN", List.of(groceryBundle, electronicsBundle));

        PricingStrategy currentPricingStrategy = new HolidayPricingStrategy();
        CostItemVisitor costVisitor = new CostItemVisitor(currentPricingStrategy);
        WeightItemVisitor weightVisitor = new WeightItemVisitor();

        shoppingCartBundle.accept(costVisitor);
        shoppingCartBundle.accept(weightVisitor);

        System.out.println("Общая стоимость покупки: " + costVisitor.getTotalCostValue());
        System.out.println("Общий вес покупки (кг): " + weightVisitor.getTotalWeightValue());
    }
}
